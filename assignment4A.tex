\section{Workflow Design}
\begin{enumerate}
    \item Quality Control and Preprocessing
    \begin{enumerate}
        \item evaluate quality of sequencing
        \\ \textbf{FastQC }
        \item trim reads
        \\ \textbf{trimmomatic}
        \item filtering
        \\ \textit{remove low quality bases from reads}
        \item Subtract Host DNA
        \\ \textit{align against human genome and remove reads that map to the host}
        \\\textbf{Bowtie2}
    \end{enumerate}
    \item Decide De Novo or Reference-based assembly
    \begin{enumerate}
        \item take a subset of sequences across all strains and map to existing reference
        \\ \textit{see if  mapping yields a high proportion of unmapped reads, low genome coverage, or lots of false SNPs}
    \end{enumerate}
    \item Generate Consensus Sequence
    \begin{enumerate}
        \item assemble viral genome 
        \item generate a consensus sequence 
        \\ \textit{only for reference-based approach}
    \end{enumerate}
    \item Validation 
    \begin{enumerate}
        \item Qualimap
    \end{enumerate}
     \\deliverable 1: \textit{fasta file with the assembled genome for each sequenced strain}
     \\deliverable 3: \textit{text file containing the average sequence variability for each window of length x}
     \\deliverable 4:  \textit{plot showing the variability along the sequenced genome}
    \item Phylogenetic Tree
    \begin{enumerate}
        \item create phylogenetic tree
    \end{enumerate}
    \\deliverable 2: \textit{phylogenetic tree in Newick format and visualised in pdf or png format.}
    
\end{enumerate}
\section{Deliverables}
